 2  #include <bits/stdc++.h>
 3  using namespace std;
 4  #define optimize() ios_base::sync_with_stdio;cin.tie(0);cout.tie(0);
 5  #define endl '\n'
 6  
 7  class Graph {
 8  private:
 9      int V; // Number of vertices
10      vector<vector<int>> adj; // Adjacency list
11  
12  public:
13      Graph(int vertices) : V(vertices), adj(vertices) {}
14  
15      void addEdge(int u, int v) {
16          adj[u].push_back(v);
17      }
18  
19      vector<int> topologicalSort() {
20          vector<int> inDegree(V, 0);
21  
22          // Calculate in-degrees for all vertices
23          for (int u = 0; u < V; u++) {
24              for (int v : adj[u]) {
25                  inDegree[v]++;
26              }
27          }
28  
29          queue<int> q;
30          for (int u = 0; u < V; u++) {
31              if (inDegree[u] == 0) {
32                  q.push(u);
33              }
34          }
35  
36          vector<int> result;
37          while (!q.empty()) {
38              int u = q.front();
39              q.pop();
40              result.push_back(u);
41  
42              for (int v : adj[u]) {
43                  if (--inDegree[v] == 0) {
44                      q.push(v);
45                  }
46              }
47          }
48  
49          if (result.size() != V) {
50              cout << "The graph has a cycle, topological sort is not possible." << endl;
51              return vector<int>();
52          }
53  
54          return result;
55      }
56  };
57  
58  int main() {
59      optimize();
60  
61      int V, E; // Number of vertices and edges
62      cout << "Enter the number of vertices and edges: ";
63      cin >> V >> E;
64  
65      Graph graph(V);
66  
67      cout << "Enter the edges (u v):" << endl;
68      for (int i = 0; i < E; i++) {
69          int u, v;
70          cin >> u >> v;
71          graph.addEdge(u, v);
72      }
73  
74      vector<int> topologicalOrder = graph.topologicalSort();
75  
76      if (!topologicalOrder.empty()) {
77          cout << "Topological Sort Order: ";
78          for (int vertex : topologicalOrder) {
79              cout << vertex << " ";
80          }
81          cout << endl;
82      }
83  
84      return 0;
85  }
